

import random
def name_to_number(name):
    if name == "rock":
        return 0
    if name == "Spock":
        return 1
    if name == "paper":
        return 2
    if name == "lizard":
        return 3
    if name == "scissors":
        return 4
    else:
        print "Error: Invalid Name"

def number_to_name(number):
     if number == 0:

     if number == 2:

        return "lizard"
